import { Injectable } from '@angular/core';
import { HttpClient, HttpErrorResponse, HttpHeaders, HttpParams } from '@angular/common/http';
import { catchError } from 'rxjs/operators';
import { Observable, throwError } from 'rxjs';
import { NbToastrService, NbComponentStatus } from '@nebular/theme';
import { API } from '../API';

/**
 * This is the main service that interacts with the API backend for competing created in DRF.
 * The API endpoints are maintained in a separate API.ts file that we are abstracting here
 * Methods for different features of the backend are implemented here.
 */
@Injectable({
  providedIn: 'root'
})
export class statementsFetchService {

  problemsURL = API.ServerURL + API.GetProblems;
  problemURL = API.ServerURL + API.GetProblem;
  contestsURL = API.ServerURL + API.GetContests;
  submitURL = API.ServerURL + API.SubmitProblem;
  submissionURL = API.ServerURL + API.GetSubmissions;
  scoreURL = API.ServerURL + API.GetScores;
  contestScoreURL = API.ServerURL + API.GetContestScores;
  contestURL = API.ServerURL + API.GetContest;
  /**
   *
   * @param toastrService Nebular Service used to pass messages as toasts
   * @param http HTTP service used to make HTTP requests
   */
  constructor(public toastrService: NbToastrService, private http: HttpClient) { }

  /**
   * This method handles errors that the API may raise by passing them as Nebular Toasts
   * @param error HTTP Error Message given by the HTTP Service
   * @param tserve Toaster Service that will be used to generate the messages
   */
  handleError(error: HttpErrorResponse, tserve: NbToastrService) {
    let status: NbComponentStatus ="danger";
    console.log(tserve)
    if (error.error instanceof ErrorEvent) {
      console.error('An error occurred:', error.error.message);
      tserve.show(error.error.message, "Error", {status})
    } else {
      console.error('An error occurred:',error.error.message);
      tserve.show(error.error.message, "Error", {status})
    }
    return throwError(
      error.message || 'Something bad happened; please try again later.');
  }

  /**
   * Service method to get problem list
   * @param data Inteface that will be passed to the API
   */
  getProblems(data): any {
    return this.http.post(this.problemsURL, data)
    .pipe(
      catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
    )
  }
  /**
   * Service method to get contests list
   * @param data Data reperesenting current state
   */
  getContests(data): any {
    return this.http.post(this.contestsURL, data)
    .pipe(
      catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
    )
  }
  /**
   * Service method to get more information about a contest
   * @param data Contest Interface (holds information about which contest to query)
   */
  getContest(data): any {
    return this.http.post(this.contestURL, data)
    .pipe(
      catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
    )
  }

  /**
   * Gets statement of the problem whose pid is given
   * @param pid Problem ID which we want to query
   */
  getProblemStatement(pid): any {
    return this.http.post(this.problemURL, {id: pid}).pipe(
      catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
    )
  }
  /**
   * Service method to submit a submission of a particular problem
   * @param data Submission data
   */
  submitProblem(data): any {
    return this.http.post(this.submitURL, data).pipe(
      catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
    )
  }
  /**
   * Service method to get past submissions made in a particular problem
   * @param data Problem Inteface
   */
  getSubmissions(data): any {
    return this.http.post(this.submissionURL, data).pipe(
      catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
    )
  }
  /**
   * Retrieves the score of a user in a particular problem
   * @param data Information about which problem and user to query for
   */
  getScores(data): any {
    return this.http.post(this.scoreURL, data).pipe(
      catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
    )
  }
    /**
   * Retrieves the score of a user in a particular contest
   * @param data Information about which contest and user to query for
   */
  getContestScores(data): any {
    return this.http.post(this.contestScoreURL, data).pipe(
      catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
    )
  }
}
