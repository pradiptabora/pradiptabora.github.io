import { HttpClient, HttpErrorResponse, HttpHeaders, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable, Subject, throwError } from 'rxjs';
import { catchError } from 'rxjs/operators';
import { fileInterface, TreeNode } from './fileInterface';
import { API } from '../API';
import { ActivatedRoute, Router } from '@angular/router';
import { NbToastrService, NbComponentStatus, NbDialogService } from '@nebular/theme';
import { map } from 'rxjs/internal/operators';
import { ConfirmDialog } from './confirmdialog/confirmdialog.component';

/**
 * Service used to connect the frontend to the backend. This includes all the functions that are used to interact with the file system present on the backend.
 */
@Injectable({
  providedIn: 'root'
})
export class CodefetchService {
  private structURL = API.ServerURL + API.GetStructure;
  private createURL = API.ServerURL + API.CreateFile;
  private codeURL = API.ServerURL + API.ReadFile;
  private deleteUrl = API.ServerURL + API.DeleteFile;
  private folderURL = API.ServerURL + API.GetFolders;
  private updateURL = API.ServerURL + API.UpdateFile;
  private filesURL = API.ServerURL + API.GetFiles;

  openedFile: fileInterface = {name: 'Choose File', id: -1, path: ''};
  openedFileData: string = ' ';
  openedFileChange: Subject<fileInterface> = new Subject<fileInterface>()
  openedFileDataChange: Subject<string> = new Subject<string>();
  /**
   * 
   * @param toastrService Used to display errors on the screen
   * @param http Used for making requests to the backend
   * @param _dialogService Used in delete files to ask for confirmation
   */
  constructor(public toastrService: NbToastrService,
              private http: HttpClient,
              private router: Router,
              private activerouter: ActivatedRoute,
              private _dialogService: NbDialogService) {
        this.openedFileChange.subscribe((value) => {
          this.openedFile = value;
        });
        this.openedFileDataChange.subscribe((value) => {
          this.openedFileData = value;
        });
  }

  /**
   * Changes the current active file to f
   * @param{fileInterface} f The file which has to be made active
   */
  changeOpenedFile(f: fileInterface) {
    this.openedFileChange.next(f);
  }

  /**
   * A common function to handle all the errors while interacting with backend
   * @param error The error object
   * @param tserve The toastr Service which is used to display errors
   */
  handleError(error: HttpErrorResponse, tserve: NbToastrService) {
    let status: NbComponentStatus ="danger";
    console.log(tserve)
    if (error.error instanceof ErrorEvent) {
      console.error('An error occurred:', error.error.message);
      tserve.show(error.error.message, "Error", {status})
    } else {
      console.error('An error occurred:',error.error.message);
      tserve.show(error.error.message, "Error", {status})
    }
    return throwError(
      error.message || 'Something bad happened; please try again later.');
  }

  /**
   * Get the list of all the files and directories belonging to the current user. 
   * @returns An Observable containing the error or the file list. Castable to the FileInterface
   */
  getFileList(): any {
    return this.http.get(this.structURL)
    .pipe(
      catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
    )
  }

  /**
   * Get the list of all the files(No folders/directories) belonging to the current user.
   * @returns Observable containing the list of files
   */
  getFiles(): any {
    return this.http.get(this.filesURL)
    .pipe(
      catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
    )
  }

  /**
   * Read the data of the file given by the fileid = id
   * @param id File Id of the file to be read
   * @returns Observable containing the data of the file. TypeCastable to CodeInterface
   */
  readfiledata(id: string): any {
    return this.http.post(this.codeURL,{file_id: id}).pipe(
      catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
    )
  }

  /**
   * Uses the [readfiledata]{@link CodefetchService#readfiledata} function to read file data and then change the active files
   * @param id Id of the file to be read
   */
  readfile(id: string): any{
    // this.router.navigate(['/dashboard/editor', {id: id}]);
    if (id == '-1') {
      this.openedFileDataChange.next("");
      return;
    }
    return this.readfiledata(id).subscribe(
      _data => {
        if(_data["success"]==false) {
          this.handleError(_data["message"], this.toastrService)
        }
        else {
          console.log(_data["data"]+"A")
          if(_data["data"]==""){
            console.log(_data["data"]+"A")
            this.openedFileDataChange.next("");
          }
          else
            this.openedFileDataChange.next(_data["data"]);
          // this._codestore.setcode(id, _data["data"]);
          // console.log(id);
        }
      }
    )
  }

  /**
   * This function is used to update the data in the files. Called when a file is saved.
   * @param id File Id of the file which is updated
   * @param data The new content of the file
   */
  updatefile(id: string, data: string): any {
    return this.http.post(this.updateURL, {file_id: id, data: data}).pipe(
      catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
    ).subscribe(
      _data => {
        if(_data["success"]==false) {
          this.handleError(_data["message"], this.toastrService);
        }
        else {
        console.log(_data["message"]);
        let status: NbComponentStatus ="success";
        this.openedFileDataChange.next(data)
        this.toastrService.show("Saved successfully", "Saved!", {status})
        }
    }
    )
  }

  /**
   * Function used to rename a file
   * @param id File Id of the file which is renamed
   * @param name New name of the file
   * @returns Boolean Promise whether the name change was successful or not
   */
  renamefile(id: string, name: string): Promise<any> {
    return this.http.post(this.updateURL, {file_id: id, filename: name}).pipe(
      catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
    ).pipe(map(
      _data => {
        if(_data["success"]==false) {
          this.handleError(_data["message"], this.toastrService);
          return false;
        }
        else {
          // window.location.reload();
          console.log(_data["message"]);
          return true;
        }
    }
    )).toPromise();
  }

  /**
   * Function used to delete a file. Caution: The file is permanently deleted from the server and can't be recovered
   * @param id File Id of the file which is deleted
   * @returns Promise which tells whether the file was deleted successfully or not
   */
  deletefile(id: string): Promise<any> {
    return this._dialogService.open(ConfirmDialog, {context: "d"}).onClose.pipe(map(res => {
      if (res) {
        return this.http.post(this.deleteUrl, {file_id: id}).pipe(
          catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
        ).pipe(map(_data => {
            if(_data["success"]==false) {
              this.handleError(_data["message"], this.toastrService);
              let status:NbComponentStatus = "danger";
              this.toastrService.show(_data["message"], "Error", {status});
              return false;
            }
            else {
              // window.location.reload();
              console.log(_data);
              return true;
            }
          }
        )).toPromise();
      }
      else {
        return false;
      }
    })).toPromise();
  }

  /**
   * Function to create new files and folders
   * @param parent Parent directory in which the file is to be created
   * @param name Name of the new File/Folder
   * @param isFile Whether the Object to be created is File or Folder
   * @returns Promise of the file ID if successful
   */
  createFile(parent: string, name: string, isFile: boolean): Promise<any> {
      var data = {
        'filename': name,
        'parent': parent,
        'is_file': isFile,

      }
      return this.http.post(this.createURL,data).pipe(
        catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
      ).pipe(map(_data => {
        if(_data["success"]==false) {
          this.handleError(_data["message"], this.toastrService);
          return false;
        }
        else {
        // window.location.reload();
          console.log(_data);
          return _data["file_id"];
        }
      })).toPromise();
  }

  /**
   * Get the list of all the folders and subfolders(No files) belonging to the current user.
   * @returns An Observable containing the error or the folder list. Castable to the FileInterface
   */
  getFolder() {
    return this.http.get(this.folderURL)
    .pipe(
      catchError((error):any => {this.handleError(error, this.toastrService)}).bind(this)
    )
  }
}
