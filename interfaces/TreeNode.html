import { StringifyOptions } from 'querystring';

/**
 * Inteface that represents a file in the filelist.
 */
export interface fileInterface {
	name?: string,
    size?: string,
    id?: number,
    is_file?: boolean,
    path?: string
}
/**
 * Interface that represents a problem statement
 */
export interface statementInterface {
  id: number,
	name: string,
  tlimit: number,
  mlimit: number
}
/**
 * Interface that represents a submission made by a user
 */
export interface submissionInterface {
  Passed: string,
	Status: string,
  Time: string
}
/**
 * Interface standing for the Score of a user in a problem/contest
 */
export interface scoreInterface {
  Score: number,
  Name: string
}
/**
 * Tree node interface used to represent entries in the Nebular Tree Grid
 */
export interface TreeNode<T> {
  data: T;
  children?: TreeNode<T>[];
  expanded?: boolean;
}
